yum安装mysql-server

yum install mysql-server mysql
vi /etc/my.cnf
    [client]
    default-character-set=utf8
service mysqld start
chkconfig mysqld on
mysql
grant all privileges on *.* to root@'localhost' identified by 'root';
flush privileges;
